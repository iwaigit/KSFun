    <script>
        document.addEventListener('DOMContentLoaded', function() {
            
            const mainContent = document.getElementById('main-content');
            const ageGate = document.getElementById('age-gate');
            const verifyAgeBtn = document.getElementById('verify-age-btn');
            const ageError = document.getElementById('age-error');

            const mobileMenuBtn = document.getElementById('mobile-menu-btn');
            const mobileMenu = document.getElementById('mobile-menu');

            let cart = [];
            let products = [];

            function init() {
                checkSessionAgeVerification();
                setupEventListeners();
                loadData();
                updateNavLinks('home');
            }

            function checkSessionAgeVerification() {
                if (sessionStorage.getItem('isAgeVerified') === 'true') {
                    ageGate.classList.add('hidden');
                    mainContent.classList.remove('hidden');
                }
            }

            function setupEventListeners() {
                verifyAgeBtn.addEventListener('click', handleAgeVerification);
                mobileMenuBtn.addEventListener('click', () => mobileMenu.classList.toggle('hidden'));
                document.querySelectorAll('.nav-link').forEach(link => {
                    link.addEventListener('click', (e) => {
                        e.preventDefault();
                        showPage(link.dataset.page);
                    });
                });
            }

            function handleAgeVerification() {
                const day = parseInt(document.getElementById('dob-day').value);
                const month = parseInt(document.getElementById('dob-month').value);
                const year = parseInt(document.getElementById('dob-year').value);
                ageError.textContent = '';

                if (isNaN(day) || isNaN(month) || isNaN(year) || day < 1 || day > 31 || month < 1 || month > 12 || year < 1900 || year > new Date().getFullYear()) {
                    ageError.textContent = 'Por favor, introduce una fecha válida.';
                    return;
                }

                const birthDate = new Date(year, month - 1, day);
                const today = new Date();
                let age = today.getFullYear() - birthDate.getFullYear();
                const m = today.getMonth() - birthDate.getMonth();
                if (m < 0 || (m === 0 && today.getDate() < birthDate.getDate())) {
                    age--;
                }

                if (age >= 21) {
                    sessionStorage.setItem('isAgeVerified', 'true');
                    ageGate.style.opacity = 0;
                    ageGate.style.visibility = 'hidden';
                    setTimeout(() => {
                        ageGate.classList.add('hidden');
                        mainContent.classList.remove('hidden');
                    }, 300);
                } else {
                    ageError.textContent = 'Debes ser mayor de 21 años para acceder.';
                }
            }
            
            function loadData() {
                const galleryGrid = document.getElementById('gallery-grid');
                for (let i = 1; i <= 12; i++) {
                    const imgUrl = `https://placehold.co/400x400/A2D2FF/FFFFFF?text=Foto+${i}`;
                    const div = document.createElement('div');
                    div.className = 'aspect-square bg-gray-200 rounded-lg overflow-hidden cursor-pointer card-zoom';
                    div.innerHTML = `<img src="${imgUrl}" alt="Foto de galería ${i}" class="w-full h-full object-cover" onclick="openImageModal('${imgUrl}')">`;
                    galleryGrid.appendChild(div);
                }

                const lingerieData = [
                    { id: 1, name: 'Conjunto Pasión', price: 49.99, sizes: ['S', 'M', 'L'], img: 'https://placehold.co/600x600/FF80AB/FFFFFF?text=Lenceria+1' },
                    { id: 2, name: 'Body Encaje', price: 59.99, sizes: ['S', 'M', 'L', 'XL'], img: 'https://placehold.co/600x600/FF80AB/FFFFFF?text=Lenceria+2' },
                    { id: 3, name: 'Bralette Seducción', price: 35.50, sizes: ['S', 'M'], img: 'https://placehold.co/600x600/FF80AB/FFFFFF?text=Lenceria+3' },
                    { id: 4, name: 'Kimono de Seda', price: 75.00, sizes: ['Unitalla'], img: 'https://placehold.co/600x600/FF80AB/FFFFFF?text=Lenceria+4' },
                ];

                const toysData = [
                    { id: 5, name: 'Vibrador Clásico', price: 69.99, sizes: [], img: 'https://placehold.co/600x600/BDE0FE/333333?text=Juguete+1' },
                    { id: 6, name: 'Anillo Vibrador', price: 29.99, sizes: [], img: 'https://placehold.co/600x600/BDE0FE/333333?text=Juguete+2' },
                    { id: 7, name: 'Bolas Chinas', price: 45.00, sizes: [], img: 'https://placehold.co/600x600/BDE0FE/333333?text=Juguete+3' },
                    { id: 8, name: 'Succionador de Clítoris', price: 89.90, sizes: [], img: 'https://placehold.co/600x600/BDE0FE/333333?text=Juguete+4' },
                ];
                
                products = [...lingerieData, ...toysData];
                
                populateShop('lingerie-grid', lingerieData);
                populateShop('toys-grid', toysData);
            }

            function populateShop(gridId, data) {
                const grid = document.getElementById(gridId);
                data.forEach(item => {
                    const card = document.createElement('div');
                    card.className = 'bg-white rounded-xl shadow-lg overflow-hidden card-zoom cursor-pointer';
                    card.innerHTML = `
                        <div class="aspect-w-1 aspect-h-1">
                            <img src="${item.img}" alt="${item.name}" class="w-full h-full object-cover">
                        </div>
                        <div class="p-4 text-center">
                            <h3 class="font-semibold text-lg">${item.name}</h3>
                            <p class="text-primary font-bold text-xl mt-1">$${item.price.toFixed(2)}</p>
                        </div>
                    `;
                    card.addEventListener('click', () => openProductModal(item.id));
                    grid.appendChild(card);
                });
            }

            window.showPage = (pageId) => {
                document.querySelectorAll('main section').forEach(section => {
                    section.classList.remove('active');
                });
                document.getElementById(pageId).classList.add('active');
                updateNavLinks(pageId);
                window.scrollTo(0, 0);
                if(mobileMenu.classList.contains('hidden') === false) {
                    mobileMenu.classList.add('hidden');
                }
            }

            function updateNavLinks(activePage) {
                document.querySelectorAll('.nav-link').forEach(link => {
                    if (link.dataset.page === activePage) {
                        link.classList.add('active', 'text-primary');
                    } else {
                        link.classList.remove('active', 'text-primary');
                    }
                });
            }

            window.openImageModal = (src) => {
                document.getElementById('modal-img').src = src;
                const modal = document.getElementById('image-modal');
                modal.classList.remove('hidden');
                modal.classList.add('flex');
            }

            window.openProductModal = (productId) => {
                const product = products.find(p => p.id === productId);
                if (!product) return;

                document.getElementById('product-modal-img').src = product.img;
                document.getElementById('product-modal-name').textContent = product.name;
                document.getElementById('product-modal-price').textContent = `$${product.price.toFixed(2)}`;
                document.getElementById('product-modal-desc').textContent = `Descubre el placer con ${product.name}. Un diseño único para momentos inolvidables. Fabricado con materiales de alta calidad, seguros para el cuerpo.`;
                
                const sizesContainer = document.getElementById('product-modal-sizes');
                if (product.sizes.length > 0) {
                    sizesContainer.classList.remove('hidden');
                    const sizeButtons = sizesContainer.querySelector('div');
                    sizeButtons.innerHTML = '';
                    product.sizes.forEach(size => {
                        const button = document.createElement('button');
                        button.className = 'border-2 border-gray-300 rounded px-3 py-1 hover:border-primary focus:border-primary focus:bg-primary/20';
                        button.textContent = size;
                        sizeButtons.appendChild(button);
                    });
                } else {
                    sizesContainer.classList.add('hidden');
                }

                document.getElementById('add-to-cart-btn').onclick = () => {
                    addToCart(productId);
                    closeModal('product-modal');
                };

                const modal = document.getElementById('product-modal');
                modal.classList.remove('hidden');
                modal.classList.add('flex');
            }

            window.closeModal = (modalId) => {
                document.getElementById(modalId).classList.add('hidden');
                document.getElementById(modalId).classList.remove('flex');
            }
            
            function addToCart(productId) {
                const existingItem = cart.find(item => item.id === productId);
                if (existingItem) {
                    existingItem.quantity++;
                } else {
                    const product = products.find(p => p.id === productId);
                    cart.push({ ...product, quantity: 1 });
                }
                updateCartUI();
            }

            function updateCartUI() {
                const cartCount = document.getElementById('cart-count');
                const cartItemsContainer = document.getElementById('cart-items');
                const cartTotalEl = document.getElementById('cart-total');

                let totalItems = 0;
                let totalPrice = 0;
                
                cartItemsContainer.innerHTML = '';

                if (cart.length === 0) {
                    cartItemsContainer.innerHTML = '<p class="text-gray-500 text-center">Tu carrito está vacío.</p>';
                } else {
                    cart.forEach(item => {
                        totalItems += item.quantity;
                        totalPrice += item.price * item.quantity;
                        const itemEl = document.createElement('div');
                        itemEl.className = 'flex items-center justify-between py-3';
                        itemEl.innerHTML = `
                            <div class="flex items-center">
                                <img src="${item.img}" class="w-16 h-16 object-cover rounded-md mr-4">
                                <div>
                                    <p class="font-semibold">${item.name}</p>
                                    <p class="text-sm text-gray-500">$${item.price.toFixed(2)}</p>
                                </div>
                            </div>
                            <div class="flex items-center">
                                <span class="mx-4">x ${item.quantity}</span>
                            </div>
                        `;
                        cartItemsContainer.appendChild(itemEl);
                    });
                }

                cartCount.textContent = totalItems;
                cartTotalEl.textContent = `$${totalPrice.toFixed(2)}`;
            }

            window.showCartModal = () => {
                const modal = document.getElementById('cart-modal');
                modal.classList.remove('hidden');
                modal.classList.add('flex');
            }

            init();
        });
    </script>
