    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const form = document.getElementById('contactForm');
            const emailInput = document.getElementById('email');
            const emailError = document.getElementById('emailError');
            const messageContainer = document.getElementById('messageContainer');

            // Referencias para el campo Nombre y su contador
            const nombreInput = document.getElementById('nombre');
            const nombreCharCount = document.getElementById('nombreCharCount');
            const maxNombreLength = 40;

            // Referencias para el campo Teléfono y Prefijo
            const prefijoInput = document.getElementById('prefijo');
            const telefonoInput = document.getElementById('telefono');

            // Referencias para el campo Edad (Fecha de Nacimiento) y el nuevo campo oculto
            const edadInput = document.getElementById('edad');
            const edadError = document.getElementById('edadError');
            const edadMinError = document.getElementById('edadMinError'); // Nuevo: mensaje de error para edad mínima
            const edadUsuarioHiddenInput = document.getElementById('edad_usuario');

            // Referencias para la nueva función de Gemini API
            const generateMessageBtn = document.getElementById('generateMessageBtn');
            const generateMessageText = document.getElementById('generateMessageText');
            const generateMessageSpinner = document.getElementById('generateMessageSpinner');
            const generatedMessageContainer = document.getElementById('generatedMessageContainer');
            const generatedMessage = document.getElementById('generatedMessage');
            const copyMessageBtn = document.getElementById('copyMessageBtn');

            // Nuevas referencias para el campo Comentario Adicional y su contador
            const comentarioAdicionalInput = document.getElementById('comentarioAdicional');
            const comentarioCharCount = document.getElementById('comentarioCharCount');
            const maxComentarioLength = 600;

            // Función para actualizar el contador de caracteres del nombre
            function updateNombreCharCount() {
                const currentLength = nombreInput.value.length;
                nombreCharCount.textContent = `${currentLength}/${maxNombreLength} caracteres`;
            }

            // Función para actualizar el contador de caracteres del comentario adicional
            function updateComentarioCharCount() {
                const currentLength = comentarioAdicionalInput.value.length;
                comentarioCharCount.textContent = `${currentLength}/${maxComentarioLength} caracteres`;
            }

            // Función para calcular la edad
            function calculateAge(dobString) {
                const parts = dobString.split('/');
                if (parts.length !== 3) return null; // Formato incorrecto

                const day = parseInt(parts[0], 10);
                const month = parseInt(parts[1], 10) - 1; // Meses son 0-indexados en JS
                const year = parseInt(parts[2], 10);

                const birthDate = new Date(year, month, day);
                const today = new Date();

                let age = today.getFullYear() - birthDate.getFullYear();
                const m = today.getMonth() - birthDate.getMonth();

                if (m < 0 || (m === 0 && today.getDate() < birthDate.getDate())) {
                    age--;
                }
                return age;
            }

            // Inicializar contadores al cargar la página
            updateNombreCharCount();
            updateComentarioCharCount();

            // Escuchar el evento 'input' para actualizar el contador del nombre en tiempo real
            nombreInput.addEventListener('input', updateNombreCharCount);

            // Escuchar el evento 'input' para actualizar el contador del comentario adicional en tiempo real
            comentarioAdicionalInput.addEventListener('input', updateComentarioCharCount);

            // Función para mostrar mensajes al usuario
            function showMessage(message, type) {
                messageContainer.textContent = message;
                messageContainer.className = `message-box ${type}`;
                messageContainer.classList.remove('hidden');
                setTimeout(() => {
                    messageContainer.classList.add('hidden');
                }, 5000);
            }

            // Validación de correo electrónico en tiempo real
            emailInput.addEventListener('input', () => {
                if (emailInput.validity.valid) {
                    emailError.classList.add('hidden');
                } else {
                    emailError.classList.remove('hidden');
                }
            });

            // Validación de formato de fecha dd/mm/aaaa y edad mínima para el campo Edad
            edadInput.addEventListener('input', () => {
                const datePattern = /^\d{2}\/\d{2}\/\d{4}$/;
                edadError.classList.add('hidden'); // Ocultar errores previos
                edadMinError.classList.add('hidden');

                if (datePattern.test(edadInput.value)) {
                    const age = calculateAge(edadInput.value);
                    if (age !== null && age >= 21) {
                        edadError.classList.add('hidden');
                        edadMinError.classList.add('hidden');
                    } else if (age !== null && age < 21) {
                        edadMinError.classList.remove('hidden');
                    } else {
                        edadError.classList.remove('hidden');
                    }
                } else {
                    edadError.classList.remove('hidden');
                }
            });

            // Función para copiar el mensaje generado al portapapeles
            copyMessageBtn.addEventListener('click', () => {
                generatedMessage.select();
                document.execCommand('copy');
                showMessage('¡Mensaje copiado al portapapeles!', 'success');
            });

            // Lógica para la generación de mensajes con Gemini API
            generateMessageBtn.addEventListener('click', async () => {
                const selectedInterest = document.querySelector('input[name="intereses"]:checked');

                if (!selectedInterest) {
                    showMessage('Por favor, selecciona un interés primero.', 'error');
                    return;
                }

                const interestValue = selectedInterest.value;
                let prompt = '';

                // Definir el prompt basado en el interés seleccionado
                switch (interestValue) {
                    case 'Presencial Hotel':
                        prompt = 'Genera un mensaje corto, seductor y discreto para una invitación a un encuentro presencial en un hotel, enfocado en el placer y la intimidad. No uses lenguaje explícito.';
                        break;
                    case 'Presencial Domicilio':
                        prompt = 'Crea un mensaje breve y atractivo para una invitación a un encuentro presencial a domicilio, mencionando sutilmente que hay "condiciones especiales" y que se discutirán en privado. No uses lenguaje explícito.';
                        break;
                    case 'Virtual (Fotos / Videos / SexyChat / a Live)':
                        prompt = 'Redacta un mensaje conciso y excitante para una invitación a una interacción virtual (fotos, videos, sexychat, live), destacando la diversión, la conexión online y la fantasía. No uses lenguaje explícito.';
                        break;
                    default:
                        prompt = 'Genera un mensaje corto y atractivo para una invitación general, destacando la emoción y la conexión.';
                }

                // Mostrar spinner y deshabilitar botón
                generateMessageText.classList.add('hidden');
                generateMessageSpinner.classList.remove('hidden');
                generateMessageBtn.disabled = true;
                generatedMessage.value = ''; // Limpiar mensaje anterior
                generatedMessageContainer.classList.add('hidden'); // Ocultar el contenedor del mensaje

                try {
                    let chatHistory = [];
                    chatHistory.push({ role: "user", parts: [{ text: prompt }] });
                    const payload = { contents: chatHistory };
                    const apiKey = ""; // La clave API será proporcionada por el entorno Canvas
                    const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

                    const response = await fetch(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    const result = await response.json();

                    if (result.candidates && result.candidates.length > 0 &&
                        result.candidates[0].content && result.candidates[0].content.parts &&
                        result.candidates[0].content.parts.length > 0) {
                        const text = result.candidates[0].content.parts[0].text;
                        generatedMessage.value = text;
                        generatedMessageContainer.classList.remove('hidden'); // Mostrar el contenedor del mensaje
                    } else {
                        showMessage('No se pudo generar el mensaje. Inténtalo de nuevo.', 'error');
                        console.error('Respuesta inesperada de la API de Gemini:', result);
                    }
                } catch (error) {
                    console.error('Error al llamar a la API de Gemini:', error);
                    showMessage(`Error al generar mensaje: ${error.message}`, 'error');
                } finally {
                    // Ocultar spinner y habilitar botón
                    generateMessageText.classList.remove('hidden');
                    generateMessageSpinner.classList.add('hidden');
                    generateMessageBtn.disabled = false;
                }
            });


            form.addEventListener('submit', async (event) => {
                event.preventDefault();

                // Validar correo electrónico
                if (!emailInput.validity.valid) {
                    emailError.classList.remove('hidden');
                    showMessage('Por favor, corrige los errores en el formulario.', 'error');
                    return;
                }

                // Validar formato de fecha y edad mínima
                const datePattern = /^\d{2}\/\d{2}\/\d{4}$/;
                if (!datePattern.test(edadInput.value)) {
                    edadError.classList.remove('hidden');
                    showMessage('Por favor, introduce una fecha válida en formato dd/mm/aaaa.', 'error');
                    return;
                }

                const age = calculateAge(edadInput.value);
                if (age === null || age < 21) {
                    edadMinError.classList.remove('hidden');
                    showMessage('Debes tener al menos 21 años para registrarte.', 'error');
                    return;
                }

                // Asignar el valor de la edad calculada al campo oculto
                edadUsuarioHiddenInput.value = age.toString(); // Convertir la edad a string

                // Recopilar los datos del formulario
                const formData = new FormData(form);
                const data = {};
                for (let [key, value] of formData.entries()) {
                    data[key] = value;
                }

                // Concatenar prefijo y teléfono para el envío
                data.telefonoCompleto = `${prefijoInput.value}${telefonoInput.value}`;
                // Puedes decidir si quieres eliminar data.prefijo o mantenerlo si el webhook lo necesita.
                // delete data.prefijo;

                // URL del webhook
                const webhookUrl = 'https://iwai-n8n-n8n.wuw7hm.easypanel.host/webhook-test/formulario';

                try {
                    const response = await fetch(webhookUrl, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Accept': 'application/json'
                        },
                        body: JSON.stringify(data)
                    });

                    if (response.ok) {
                        showMessage('¡Formulario enviado con éxito!', 'success');
                        form.reset();
                        emailError.classList.add('hidden');
                        edadError.classList.add('hidden'); // Ocultar error de edad al resetear
                        edadMinError.classList.add('hidden'); // Ocultar error de edad mínima
                        updateNombreCharCount(); // Reinicia el contador de caracteres del nombre
                        updateComentarioCharCount(); // Reinicia el contador de caracteres del comentario
                        generatedMessageContainer.classList.add('hidden'); // Ocultar el mensaje generado
                        generatedMessage.value = ''; // Limpiar el textarea del mensaje generado
                    } else {
                        const errorText = await response.text();
                        console.error('Error al enviar el formulario:', response.status, errorText);
                        showMessage(`Error al enviar el formulario: ${response.status} - ${errorText || 'Respuesta inesperada'}`, 'error');
                    }
                } catch (error) {
                    console.error('Error en la solicitud fetch:', error);
                    showMessage(`Error de conexión: ${error.message}`, 'error');
                }
            });
        });
    </script>
